## 🚀 解决方案：使用Word文档提取

### 实施步骤

**步骤1：创建Word文档提取服务**

```javascript
const mammoth = require('mammoth');
const fs = require('fs');

/**
 * 从Word文档提取文本
 * @param {string} filePath - Word文档路径
 * @returns {Promise<string>} 提取的文本
 */
async function extractFromWord(filePath) {
  try {
    const result = await mammoth.extractRawText({ path: filePath });
    return result.value;  // 干净的文本，格式完整
  } catch (error) {
    console.error('Word文档提取失败:', error);
    throw error;
  }
}

// 使用示例
extractFromWord('学生手册.docx')
  .then(text => {
    console.log(text);
  });
```

**步骤2：更新学生手册提取脚本**

找到当前的学生手册提取脚本，修改为使用Word文档而不是PDF。

**步骤3：批量处理所有Word文档**

```javascript
const glob = require('glob');
const path = require('path');

// 查找所有.docx文件
glob('内江师范学院相关信息附件/**/*.docx', (err, files) => {
  files.forEach(file => {
    console.log('处理:', file);
    extractFromWord(file).then(text => {
      // 保存提取的文本
      const outputPath = file.replace('.docx', '.txt');
      fs.writeFileSync(outputPath, text);
    });
  });
});
```
