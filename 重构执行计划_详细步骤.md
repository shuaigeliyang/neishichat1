### 4.2 缓存服务
**文件：`services/infrastructure/cache.js`**

```javascript
const Redis = require('ioredis');
const config = require('../../config');
const logger = require('./logger');

class CacheService {
    constructor() {
        this.redis = null;
        this.memoryCache = new Map();
        this.connected = false;
        this.init();
    }

    async init() {
        try {
            this.redis = new Redis({
                host: config.cache.redis.host,
                port: config.cache.redis.port,
                password: config.cache.redis.password,
                db: config.cache.redis.db,
                retryStrategy: (times) => {
                    const delay = Math.min(times * 50, 2000);
                    return delay;
                },
                maxRetriesPerRequest: 3,
                lazyConnect: true
            });

            this.redis.on('connect', () => {
                logger.info('Redis connected');
                this.connected = true;
            });

            this.redis.on('error', (err) => {
                logger.error('Redis error:', err);
                this.connected = false;
            });

            await this.redis.connect();
        } catch (error) {
            logger.warn('Redis connection failed, using memory cache:', error.message);
            this.connected = false;
        }
    }

    async get(key) {
        try {
            if (this.connected) {
                const value = await this.redis.get(key);
                if (value) return JSON.parse(value);
            }
        } catch (error) {
            logger.error('Cache get error:', error);
        }

        // 降级到内存缓存
        return this.memoryCache.get(key);
    }

    async set(key, value, ttl = 300) {
        try {
            if (this.connected) {
                await this.redis.setex(key, ttl, JSON.stringify(value));
            }
        } catch (error) {
            logger.error('Cache set error:', error);
        }

        // 同时设置内存缓存
        this.memoryCache.set(key, value);
        
        // 自动过期内存缓存
        setTimeout(() => {
            this.memoryCache.delete(key);
        }, ttl * 1000);
    }

    async del(key) {
        try {
            if (this.connected) {
                await this.redis.del(key);
            }
        } catch (error) {
            logger.error('Cache delete error:', error);
        }

        this.memoryCache.delete(key);
    }

    async clear() {
        try {
            if (this.connected) {
                await this.redis.flushdb();
            }
        } catch (error) {
            logger.error('Cache clear error:', error);
        }

        this.memoryCache.clear();
    }

    isConnected() {
        return this.connected;
    }
}

module.exports = new CacheService();
```

```javascript
class AnthropicService {

        this.cacheEnabled = true;
    }

    async chat(message, options = {}) {
        try {
            const {
                systemPrompt = null,
                history = [],

                maxTokens = this.config.maxTokens,
                useCache = true

                }
```

**文件：`services/ai/embeddingClient.js`**（已存在，需要更新）

```javascript
const axios = require('axios');
const config = require('../../config');
const logger = require('../infrastructure/logger');
const cache = require('../infrastructure/cache');

class PythonEmbeddingClient {
    constructor() {
        this.serviceUrl = config.api.embedding.serviceUrl;
        this.timeout = config.api.embedding.timeout;
        this.model = config.api.embedding.model;
    }

    async getEmbedding(text) {
        try {
            // 检查缓存
            const cacheKey = `embedding:${Buffer.from(text).toString('base64').substring(0, 50)}`;
            const cached = await cache.get(cacheKey);
            if (cached) {
                logger.debug('Using cached embedding');
                return cached;
            }

            // 调用Python服务
            const response = await axios.post(
                this.serviceUrl + '/embed',
                { text: text },
                {
                    timeout: this.timeout,
                    headers: { 'Content-Type': 'application/json' }
                }
            );

            const embedding = response.data.embedding;

            // 验证向量维度
            if (embedding.length !== this.config.dimension) {
                throw new Error(`Invalid embedding dimension: ${embedding.length}, expected ${this.config.dimension}`);
            }

            // 缓存结果
            await cache.set(cacheKey, embedding, config.cache.redis.ttl.embedding);

            logger.debug('Embedding generated successfully', {
                dimension: embedding.length,
                cached: false
            });

            return embedding;

        } catch (error) {
            logger.error('Embedding generation failed:', {
                error: error.message,
                code: error.code
            });

            throw new Error('向量生成失败，请检查Embedding服务是否运行。');
        }
    }

    async getBatchEmbeddings(texts, batchSize = 8) {
        const embeddings = [];

        for (let i = 0; i < texts.length; i += batchSize) {
            const batch = texts.slice(i, i + batchSize);
            const response = await axios.post(
                this.serviceUrl + '/embed_batch',
                { texts: batch },
                {
                    timeout: this.timeout * batchSize,
                    headers: { 'Content-Type': 'application/json' }
                }
            );

            embeddings.push(...response.data.embeddings);
        }

        return embeddings;
    }

    cosineSimilarity(vec1, vec2) {
        if (!vec1 || !vec2 || vec1.length !== vec2.length) return 0;

        let dot = 0, norm1 = 0, norm2 = 0;
        for (let i = 0; i < vec1.length; i++) {
            dot += vec1[i] * vec2[i];
            norm1 += vec1[i] * vec1[i];
            norm2 += vec2[i] * vec2[i];
        }

        return norm1 && norm2 ? dot / (Math.sqrt(norm1) * Math.sqrt(norm2)) : 0;
    }
}

module.exports = new PythonEmbeddingClient();
```

```javascript
const path = require('path');
const fs = require('fs').promises;
const logger = require('../infrastructure/logger');
const config = require('../../config');
const anthropicService = require('./anthropicService');
const embeddingClient = require('./embeddingClient');

class RAGService {
    constructor() {
        this.indexPath = path.join(__dirname, '../../../文档库/indexes/unified_index.json');
        this.indexData = null;
        this.initialized = false;
    }

    async initialize() {
        if (this.initialized) return;

        try {
            logger.info('Initializing RAG service...');
            
            const data = await fs.readFile(this.indexPath, 'utf-8');
            this.indexData = JSON.parse(data);

            // 验证索引
            if (!this.indexData.chunks || !this.indexData.documents) {
                throw new Error('Invalid index format');
            }

            const chunksWithEmbedding = this.indexData.chunks.filter(c => c.embedding).length;
            logger.info('RAG service initialized', {
                documents: this.indexData.documents.length,
                totalChunks: this.indexData.chunks.length,
                chunksWithEmbedding
            });

            this.initialized = true;
        } catch (error) {
            logger.error('Failed to initialize RAG service:', error);
            throw error;
        }
    }

    async ask(question, options = {}) {
        await this.initialize();

        const {
            topK = 5,
            minScore = 0.3,
            documentIds = null
        } = options;

        try {
            // 1. 生成问题向量
            const questionEmbedding = await embeddingClient.getEmbedding(question);

            // 2. 检索相关chunks
            const relevantChunks = this.search(questionEmbedding, {
                topK,
                minScore,
                documentIds
            });

            if (relevantChunks.length === 0) {
                return {
                    answer: '抱歉，没有找到与您问题相关的内容。',
                    sources: [],
                    confidence: 0
                };
            }

            // 3. 构建上下文
            const context = this.buildContext(relevantChunks);

            // 4. 生成答案
            const systemPrompt = this.getSystemPrompt();
            const answer = await anthropicService.chat(question, {
                systemPrompt,
                history: [],
                useCache: true
            });

            // 5. 格式化来源
            const sources = this.formatSources(relevantChunks);

            return {
                answer,
                sources,
                confidence: this.calculateConfidence(relevantChunks)
            };

        } catch (error) {
            logger.error('RAG query failed:', error);
            throw error;
        }
    }

    search(questionEmbedding, options) {
        const { topK, minScore, documentIds } = options;

        let chunks = this.indexData.chunks;

        // 文档过滤
        if (documentIds && documentIds.length > 0) {
            chunks = chunks.filter(c => documentIds.includes(c.documentId));
        }

        // 计算相似度
        const results = chunks.map(chunk => {
            if (!chunk.embedding) {
                return { chunk, score: 0 };
            }

            const score = embeddingClient.cosineSimilarity(questionEmbedding, chunk.embedding);
            return { chunk, score };
        });

        // 过滤和排序
        return results
            .filter(r => r.score >= minScore)
            .sort((a, b) => b.score - a.score)
            .slice(0, topK);
    }

    buildContext(chunks) {
        return chunks.map((chunk, index) => {
            const doc = this.indexData.documents.find(d => d.documentId === chunk.documentId);
            return `[文档${index + 1}] ${doc.displayName || doc.name} (相关度: ${(chunk.score * 100).toFixed(1)}%)\n${chunk.text || chunk.content || chunk.full_context}`;
        }).join('\n\n');
    }

    getSystemPrompt() {
        return `你是教育系统的智能助手。

【重要规则】
1. 你只能根据下面提供的【参考文档】内容来回答问题
2. 禁止编造、推测或添加任何参考文档中没有的信息
3. 如果参考文档中没有相关信息，必须明确回答"根据提供的文档，没有找到相关内容"
4. 回答时要引用具体页码

请严格基于上述参考文档回答，不要添加任何文档中没有的内容。`;
    }

    formatSources(chunks) {
        const sourcesMap = new Map();

        chunks.forEach(({ chunk, score }) => {
            if (!sourcesMap.has(chunk.documentId)) {
                const doc = this.indexData.documents.find(d => d.documentId === chunk.documentId);
                sourcesMap.set(chunk.documentId, {
                    documentId: chunk.documentId,
                    documentName: doc.displayName || doc.name,
                    chunks: []
                });
            }

            sourcesMap.get(chunk.documentId).chunks.push({
                page: chunk.page || chunk.metadata?.page,
                score: score,
                snippet: (chunk.text || chunk.content || chunk.full_context || '').substring(0, 200)
            });
        });

        return Array.from(sourcesMap.values());
    }

    calculateConfidence(chunks) {
        if (chunks.length === 0) return 0;
        
        const avgScore = chunks.reduce((sum, c) => sum + c.score, 0) / chunks.length;
        const maxScore = chunks[0].score;

        // 综合平均分和最高分
        return (avgScore * 0.6 + maxScore * 0.4);
    }

    getStatistics() {
        if (!this.indexData) {
            return null;
        }

        return {
            totalDocuments: this.indexData.documents.length,
            totalChunks: this.indexData.chunks.length,
            chunksWithEmbedding: this.indexData.chunks.filter(c => c.embedding).length,
            documents: this.indexData.documents.map(doc => ({
                documentId: doc.documentId,
                name: doc.name,
                displayName: doc.displayName,
                status: doc.status,
                chunkCount: this.indexData.chunks.filter(c => c.documentId === doc.documentId).length
            }))
        };
    }
}

module.exports = new RAGService();
```
