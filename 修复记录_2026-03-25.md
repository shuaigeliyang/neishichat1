```javascript
/**
 * 教师导出授课班级信息
 * GET /api/export/teacher/classes
 */
router.get('/teacher/classes', authenticate, async (req, res) => {
  const { id: userId, type: userType } = req.user;

  // 只有教师才能导出授课班级
  if (userType !== 'teacher') {
    return error(res, '只有教师才能导出授课班级信息', 403);
  }

  const classes = await query(`
    SELECT
      c.class_id AS '班级ID',
      c.class_name AS '班级名称',
      m.major_name AS '专业名称',
      m.major_code AS '专业代码',
      col.college_name AS '学院名称',
      col.college_code AS '学院代码',
      COUNT(DISTINCT co.course_id) AS '授课门数',
      GROUP_CONCAT(DISTINCT course.course_name SEPARATOR '、') AS '授课课程'
    FROM course_offerings co
    LEFT JOIN classes c ON co.class_id = c.class_id
    LEFT JOIN majors m ON c.major_id = m.major_id
    LEFT JOIN colleges col ON m.college_id = col.college_id
    LEFT JOIN courses course ON co.course_id = course.course_id
    WHERE co.teacher_id = ?
      AND co.status = 1
    GROUP BY co.class_id, c.class_name, m.major_name, m.major_code, col.college_name, col.college_code
    ORDER BY c.class_name
  `, [userId]);

  // 生成Excel文件...
});
```

## 🔧 问题3：AI生成SQL别名错误

### 问题描述

**用户：** 赵燕老师

**输入：** "下载我授课班级的信息"

**AI生成的错误SQL：**
```sql
SELECT c.class_name, cl.class_id, co.semester, co.classroom, co.status
FROM course_offerings co
JOIN classes cl ON co.class_id = cl.class_id
WHERE co.teacher_id = 1 AND co.status = 1
```

**错误信息：**
```
Unknown column 'c.class_name' in 'field list'
```

**原因：** AI使用了 `c.class_name`，但JOIN中定义的别名是 `classes cl`（别名是cl不是c）

---

### 修复方案

**文件：** `backend/src/services/sqlGenerator.js`

**新增示例5和示例6：**

```sql
### 示例5：教师查询授课班级
用户："查询我授课的班级信息" 或 "下载我授课班级的信息"
SQL：
SELECT
  cl.class_id AS '班级ID',
  cl.class_name AS '班级名称',
  m.major_name AS '专业名称',
  col.college_name AS '学院名称',
  COUNT(DISTINCT co.course_id) AS '授课门数',
  GROUP_CONCAT(DISTINCT c.course_name SEPARATOR '、') AS '授课课程'
FROM course_offerings co
LEFT JOIN classes cl ON co.class_id = cl.class_id
LEFT JOIN majors m ON cl.major_id = m.major_id
LEFT JOIN colleges col ON m.college_id = col.college_id
LEFT JOIN courses c ON co.course_id = c.course_id
WHERE co.teacher_id = 1 AND co.status = 1
GROUP BY cl.class_id, cl.class_name, m.major_name, col.college_name
ORDER BY cl.class_name

### 示例6：教师查询授课课程
用户："查询我授课的课程" 或 "我的授课课程"
SQL：
SELECT
  c.course_name AS '课程名称',
  c.course_code AS '课程代码',
  cl.class_name AS '授课班级',
  co.semester AS '学期',
  co.schedule AS '上课时间',
  co.classroom AS '教室'
FROM course_offerings co
LEFT JOIN courses c ON co.course_id = c.course_id
LEFT JOIN classes cl ON co.class_id = cl.class_id
WHERE co.teacher_id = 1 AND co.status = 1
ORDER BY co.semester DESC, c.course_name
```

---

### 表别名规范

**修复后的别名使用：**
- `co` → course_offerings（课程开课表）
- `cl` → classes（班级表）
- `c` → courses（课程表）
- `m` → majors（专业表）
- `col` → colleges（学院表）

**关键规则：**
- ✅ 所有别名定义后必须保持一致
- ✅ SELECT中的别名必须与JOIN中定义的别名一致

### 数据库设计分析

**学生选课信息存储位置：**
- ❌ **没有** enrollments（选课表）
- ✅ **有** grades（成绩表），通过 offering_id 关联课程开课信息

**陈丽的选课情况：**
- grades表中有16条成绩记录
- 每条记录对应一门课（通过 offering_id 关联）

---

### 修复方案

**文件：** `backend/src/services/sqlGenerator.js`

**新增示例7：**
```sql
### 示例7：学生查询课程表（重要！）
用户："我的课程表" 或 "查询我的课程" 或 "我的课表"
SQL：
SELECT
  c.course_name AS '课程名称',
  c.course_code AS '课程代码',
  co.semester AS '学期',
  co.schedule AS '上课时间',
  co.classroom AS '教室',
  t.name AS '授课教师',
  cl.class_name AS '班级'
FROM grades g
LEFT JOIN course_offerings co ON g.offering_id = co.offering_id
LEFT JOIN courses c ON co.course_id = c.course_id
LEFT JOIN teachers t ON co.teacher_id = t.teacher_id
LEFT JOIN classes cl ON co.class_id = cl.class_id
WHERE g.student_id = 1
ORDER BY co.semester DESC, c.course_name

注意：学生查询课程表必须从 grades 表开始，通过 offering_id 关联课程信息！
```

**系统提示词中强调：**
```javascript
- 当查询"我的课程表"或"我的课表"时（非常重要！）：
  - 必须从 grades 表开始查询：FROM grades g
  - 通过 offering_id 关联：LEFT JOIN course_offerings co ON g.offering_id = co.offering_id
  - WHERE 条件：WHERE g.student_id = ${userContext.id}
  - 不要直接查询 course_offerings 表！
```

### 学生查询课程表的正确流程

**表结构：**
```
grades（成绩表）
  ├─ student_id（学生ID）
  ├─ offering_id（课程开课ID）
  └─ total_score（总评成绩）

course_offerings（课程开课表）
  ├─ offering_id（开课ID）
  ├─ course_id（课程ID）
  ├─ teacher_id（教师ID）
  ├─ semester（学期）
  ├─ schedule（上课时间）
  └─ classroom（教室）

courses（课程表）
  ├─ course_id（课程ID）
  ├─ course_name（课程名称）
  └─ course_code（课程代码）

teachers（教师表）
  ├─ teacher_id（教师ID）
  └─ name（教师姓名）

classes（班级表）
  ├─ class_id（班级ID）
  └─ class_name（班级名称）
```

**JOIN顺序：**
1. FROM grades g（主表，学生成绩）
2. LEFT JOIN course_offerings co（课程开课信息）
3. LEFT JOIN courses c（课程基本信息）
4. LEFT JOIN teachers t（授课教师）
5. LEFT JOIN classes cl（班级信息）
